# 步骤 4：生成基础 API 路由器

创建 `src/api/base.py`，定义基础 API 组件：

```python
from typing import Generic, TypeVar, List, Optional, Dict, Any
from fastapi import APIRouter, HTTPException, status, Depends
from pydantic import BaseModel

T = TypeVar("T")

class APIResponse(BaseModel, Generic[T]):
    """统一响应格式"""
    code: int = 0
    message: str = "success"
    data: Optional[T] = None

class PaginatedResponse(BaseModel, Generic[T]):
    """分页响应"""
    code: int = 0
    message: str = "success"
    data: List[T] = []
    total: int = 0
    page: int = 1
    page_size: int = 20

class BaseRouter:
    """基础路由器"""

    def __init__(self, prefix: str, tags: List[str]):
        self.router = APIRouter(prefix=prefix, tags=tags)

    def ok(self, data: Any = None, message: str = "success") -> APIResponse:
        """成功响应"""
        return APIResponse(code=0, message=message, data=data)

    def error(self, message: str, code: int = 500) -> HTTPException:
        """错误响应"""
        return HTTPException(status_code=code, detail=message)
```

# 步骤 5：生成模块 CRUD 路由

根据 API.md，在 `src/api/{模块名}/` 下创建路由文件。

**路由文件模板**：

```python
from typing import List
from fastapi import APIRouter, Depends, Query, Path
from sqlalchemy.orm import Session
from src.core.database import get_db
from src.services.{模块名} import Entity1Service
from src.schemas.{模块名} import (
    Entity1Create,
    Entity1Update,
    Entity1Response,
    Entity1ListResponse
)
from src.api.base import APIResponse
from src.api.dependencies import get_current_user  # 如果需要认证

router = APIRouter(prefix="/{模块}/entity1", tags=["Entity1"])

@router.post("/", response_model=APIResponse[Entity1Response], summary="创建Entity1")
async def create_entity1(
    data: Entity1Create,
    db: Session = Depends(get_db),
    # current_user = Depends(get_current_user)  # 如果需要认证
):
    """
    创建新的 Entity1

    - **field1**: 字段1说明
    - **field2**: 字段2说明
    """
    service = Entity1Service(db)
    result = service.create(data)
    return APIResponse(data=result)

@router.get("/{id}", response_model=APIResponse[Entity1Response], summary="获取Entity1")
async def get_entity1(
    id: str = Path(..., description="Entity1 ID"),
    db: Session = Depends(get_db)
):
    """根据 ID 获取 Entity1"""
    service = Entity1Service(db)
    result = service.get_by_id(id)
    return APIResponse(data=result)

@router.get("/", response_model=APIResponse[dict], summary="Entity1列表")
async def list_entity1(
    page: int = Query(1, ge=1, description="页码"),
    page_size: int = Query(20, ge=1, le=100, description="每页数量"),
    field1: str = Query(None, description="field1过滤"),
    db: Session = Depends(get_db)
):
    """获取 Entity1 列表"""
    service = Entity1Service(db)

    filters = {}
    if field1:
        filters["field1"] = field1

    result = service.list_all(page=page, page_size=page_size, **filters)
    return APIResponse(data=result)

@router.put("/{id}", response_model=APIResponse[Entity1Response], summary="更新Entity1")
async def update_entity1(
    id: str = Path(..., description="Entity1 ID"),
    data: Entity1Update = None,
    db: Session = Depends(get_db)
):
    """更新 Entity1"""
    service = Entity1Service(db)
    result = service.update(id, data)
    return APIResponse(data=result)

@router.delete("/{id}", response_model=APIResponse[bool], summary="删除Entity1")
async def delete_entity1(
    id: str = Path(..., description="Entity1 ID"),
    db: Session = Depends(get_db)
):
    """删除 Entity1"""
    service = Entity1Service(db)
    result = service.delete(id)
    return APIResponse(data=result)
```

# 步骤 6：生成模块间 API（内部接口）

如果模块需要向其他模块提供接口，创建 `src/api/{模块名}/internal.py`：

```python
from fastapi import APIRouter, Depends
from sqlalchemy.orm import Session
from src.core.database import get_db
from src.services.{模块名} import Entity1Service
from src.schemas.internal import InternalEntity1Request, InternalResponse

router = APIRouter(prefix="/internal/{模块}", tags=["Internal"])

@router.post("/entity1/get-by-field1", response_model=InternalResponse)
async def internal_get_entity1(
    request: InternalEntity1Request,
    db: Session = Depends(get_db)
):
    """
    内部接口：根据 field1 获取 Entity1
    供其他模块调用
    """
    service = Entity1Service(db)
    entity = service.repo.find_by_field1(request.field1)

    if not entity:
        return InternalResponse(code=404, message="Not Found")

    return InternalResponse(
        code=0,
        data={"id": entity.id, "field1": entity.field1}
    )
```

# 步骤 7：注册路由到主应用

更新 `main.py`，注册模块路由：

```python
from fastapi import FastAPI
from src.api.{模块名} import router as {模块}_router
from src.api.{模块名}.internal import router as {模块}_internal_router

app = FastAPI(title="项目名称")

# 对外 API
app.include_router({模块}_router)

# 内部 API
app.include_router({模块}_internal_router)
```

import pytest
from fastapi.testclient import TestClient

        data = response.json()

    def test_update_entity1(self, db: Session):
        """测试更新"""
