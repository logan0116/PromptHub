# 步骤 5：生成基础 Service 类

创建 `src/services/base.py`，定义基础 Service 类：

```python
from typing import Generic, TypeVar, Optional, List, Dict, Any
from sqlalchemy.orm import Session
from pydantic import BaseModel
from abc import ABC, abstractmethod

T = TypeVar("T")

class ServiceError(Exception):
    """服务异常基类"""
    def __init__(self, message: str, code: int = 500):
        self.message = message
        self.code = code
        super().__init__(message)

class ValidationError(ServiceError):
    """验证错误"""
    def __init__(self, message: str):
        super().__init__(message, code=400)

class NotFoundError(ServiceError):
    """资源不存在错误"""
    def __init__(self, resource: str = "资源"):
        super().__init__(f"{resource}不存在", code=404)

class BusinessError(ServiceError):
    """业务错误"""
    def __init__(self, message: str):
        super().__init__(message, code=422)

class BaseService(ABC, Generic[T]):
    """基础 Service 类"""

    def __init__(self, db: Session):
        self.db = db

    @abstractmethod
    def get_by_id(self, id: str) -> Optional[T]:
        """根据 ID 获取"""
        pass

    @abstractmethod
    def list_all(self, **params) -> List[T]:
        """列表查询"""
        pass

    @abstractmethod
    def create(self, data: Dict[str, Any]) -> T:
        """创建"""
        pass

    @abstractmethod
    def update(self, id: str, data: Dict[str, Any]) -> T:
        """更新"""
        pass

    @abstractmethod
    def delete(self, id: str) -> bool:
        """删除"""
        pass
```

# 步骤 6：生成 DTO/Schema 类

创建 `src/schemas/{模块名}/` 目录，定义请求/响应 Schema：

```python
from pydantic import BaseModel, Field
from typing import Optional
from datetime import datetime

class Entity1Base(BaseModel):
    """Entity1 基础 Schema"""
    field1: str = Field(..., description="字段1说明")
    field2: Optional[str] = Field(None, description="字段2说明")

class Entity1Create(Entity1Base):
    """创建请求"""
    pass

class Entity1Update(BaseModel):
    """更新请求"""
    field1: Optional[str] = None
    field2: Optional[str] = None

class Entity1Response(Entity1Base):
    """响应"""
    id: str
    created_at: datetime
    updated_at: datetime

    class Config:
        from_attributes = True

class Entity1ListResponse(BaseModel):
    """列表响应"""
    items: List[Entity1Response]
    total: int
    page: int
    page_size: int
```

# 步骤 7：生成模块 Service

根据 BUSINESS.md 和 API.md，在 `src/services/{模块名}/` 下创建 Service 文件。

**Service 文件模板**：

```python
from typing import List, Optional
from sqlalchemy.orm import Session
from src.repositories.{模块名} import Entity1Repository
from src.schemas.{模块名} import Entity1Create, Entity1Update, Entity1Response
from src.services.base import BaseService, ValidationError, NotFoundError, BusinessError

class Entity1Service(BaseService):
    """Entity1 业务逻辑"""

    def __init__(self, db: Session):
        super().__init__(db)
        self.repo = Entity1Repository(db)

    def get_by_id(self, id: str) -> Entity1Response:
        """根据 ID 获取"""
        entity = self.repo.get_by_id(id)
        if not entity:
            raise NotFoundError("Entity1")
        return Entity1Response.model_validate(entity)

    def list_all(
        self,
        page: int = 1,
        page_size: int = 20,
        **filters
    ) -> dict:
        """列表查询"""
        skip = (page - 1) * page_size
        result = self.repo.list_all(skip=skip, limit=page_size, **filters)

        return {
            "items": [Entity1Response.model_validate(item) for item in result.items],
            "total": result.total,
            "page": page,
            "page_size": page_size
        }

    def create(self, data: Entity1Create) -> Entity1Response:
        """创建"""
        # 业务规则验证
        if self.repo.find_by_field1(data.field1):
            raise ValidationError(f"field1={data.field1} 已存在")

        # 创建实体
        from src.models.{模块名} import Entity1
        entity = Entity1(**data.model_dump())
        entity = self.repo.create(entity)

        # 发布事件（如果需要）
        # self.event_publisher.publish("Entity1Created", entity.id)

        return Entity1Response.model_validate(entity)

    def update(self, id: str, data: Entity1Update) -> Entity1Response:
        """更新"""
        entity = self.repo.get_by_id(id)
        if not entity:
            raise NotFoundError("Entity1")

        # 更新字段
        update_data = data.model_dump(exclude_unset=True)
        for field, value in update_data.items():
            setattr(entity, field, value)

        entity = self.repo.update(entity)
        return Entity1Response.model_validate(entity)

    def delete(self, id: str) -> bool:
        """删除"""
        # 检查是否可删除（业务规则）
        entity = self.repo.get_by_id(id)
        if not entity:
            raise NotFoundError("Entity1")

        # TODO: 检查关联数据，防止数据一致性问题

        return self.repo.delete(id)

    # 业务方法
    def custom_business_method(self, param: str) -> dict:
        """自定义业务方法"""
        # 实现业务逻辑
        pass
```

# 步骤 9：编写 Service 单元测试

创建 `tests/services/test_{模块名}.py`：

```python
import pytest
from sqlalchemy.orm import Session
from src.services.{模块名} import Entity1Service
from src.schemas.{模块名} import Entity1Create
from src.services.base import NotFoundError, ValidationError

class TestEntity1Service:
    """测试 Entity1Service"""

    def test_create_entity(self, db: Session):
        """测试创建实体"""
        service = Entity1Service(db)
        data = Entity1Create(field1="test", field2="desc")
        result = service.create(data)

        assert result.id is not None
        assert result.field1 == "test"

    def test_create_duplicate_fails(self, db: Session):
        """测试重复创建失败"""
        service = Entity1Service(db)
        data = Entity1Create(field1="unique", field2="desc")
        service.create(data)

        with pytest.raises(ValidationError):
            service.create(data)

    def test_get_not_found(self, db: Session):
        """测试获取不存在实体"""
        service = Entity1Service(db)
        with pytest.raises(NotFoundError):
            service.get_by_id("non-existent-id")

    def test_update_entity(self, db: Session):
        """测试更新实体"""
        service = Entity1Service(db)
        data = Entity1Create(field1="test", field2="desc")
        created = service.create(data)

        from src.schemas.{模块名} import Entity1Update
        updated = service.update(created.id, Entity1Update(field2="updated"))
        assert updated.field2 == "updated"

    def test_delete_entity(self, db: Session):
        """测试删除实体"""
        service = Entity1Service(db)
        data = Entity1Create(field1="test", field2="desc")
        created = service.create(data)

        result = service.delete(created.id)
        assert result is True

        with pytest.raises(NotFoundError):
            service.get_by_id(created.id)
```
