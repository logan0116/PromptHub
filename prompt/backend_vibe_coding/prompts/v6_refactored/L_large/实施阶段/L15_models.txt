# 步骤 5：生成基础模型类

创建 `src/models/base.py`，定义基础模型类：

```python
from datetime import datetime
from uuid import uuid4
from typing import Optional
from sqlalchemy import Column, String, DateTime, Boolean
from sqlalchemy.ext.declarative import declarative_base

Base = declarative_base()

class BaseModel(Base):
    """基础模型类"""
    __abstract__ = True

    id = Column(String(36), primary_key=True, default=lambda: str(uuid4()))
    created_at = Column(DateTime, default=datetime.utcnow, nullable=False)
    updated_at = Column(DateTime, default=datetime.utcnow, onupdate=datetime.utcnow, nullable=False)
    deleted_at = Column(DateTime, nullable=True)

    def is_deleted(self) -> bool:
        return self.deleted_at is not None

    def soft_delete(self):
        self.deleted_at = datetime.utcnow()
```

# 步骤 6：生成模块模型

根据 SCHEMA.md，在 `src/models/{模块名}/` 下创建模型文件。

每个实体一个文件，文件名为 `{实体名小写}.py`。

**模型文件模板**：

```python
from sqlalchemy import Column, String, Integer, Text, ForeignKey, Index
from sqlalchemy.orm import relationship
from src.models.base import BaseModel, Base

class {实体名}(BaseModel):
    """{实体描述}"""
    __tablename__ = "{表名}"

    # 字段定义
    field1 = Column(String(255), nullable=False, comment="字段说明")
    field2 = Column(Integer, default=0, comment="字段说明")

    # 外键关系
    related_id = Column(String(36), ForeignKey("related_table.id"), nullable=True)

    # 关系
    related = relationship("RelatedModel", back_populates="this")

    # 索引
    __table_args__ = (
        Index("idx_field1", "field1"),
    )

    def __repr__(self):
        return f"<{实体名}(id={self.id}, field1={self.field1})>"
```

# 步骤 9：编写模型单元测试

创建 `tests/models/test_{模块名}.py`：

```python
import pytest
from sqlalchemy.orm import Session
from src.models.{模块名} import Entity1

class TestEntity1:
    """测试 Entity1 模型"""

    def test_create_entity(self, db: Session):
        """测试创建实体"""
        entity = Entity1(field1="test")
        db.add(entity)
        db.commit()
        db.refresh(entity)

        assert entity.id is not None
        assert entity.field1 == "test"

    def test_soft_delete(self, db: Session):
        """测试软删除"""
        entity = Entity1(field1="test")
        db.add(entity)
        db.commit()

        entity.soft_delete()
        db.commit()

        assert entity.is_deleted() is True
```
